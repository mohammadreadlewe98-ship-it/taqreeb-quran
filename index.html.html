يجب استبدال كود Apps Script القديم بهذا الكود الجديد الذي يدعم تتبع وقت البدء الحقيقي لكل طالب عبر الشيت. بعد النشر، أدخل الرابط الجديد أدناه.

كود Apps Script الكامل (انسخه واستبدل القديم)

const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); const EXAM_DURATION_MS = 30 * 60 * 1000; // 30 minutes const PASS = "000999"; function doGet(e) { const action = e.parameter.action || ""; const ss = SpreadsheetApp.openById(SHEET_ID); let result; try { if (action === "getQuestions") { const sheet = ss.getSheetByName("Questions"); const data = sheet ? sheet.getDataRange().getValues() : []; const questions = []; for (let i = 1; i < data.length; i++) { if (!data[i][0]) continue; const opts = [data[i][1],data[i][2],data[i][3],data[i][4],data[i][5],data[i][6]] .filter(o => o !== "" && o !== null && o !== undefined); const correctLetter = String(data[i][7]||data[i][5]).trim().toUpperCase(); const correctIndex = ["A","B","C","D","E","F"].indexOf(correctLetter); questions.push({ text: String(data[i][0]), options: opts, correct: correctIndex >= 0 ? correctIndex : 0 }); } result = { status:"ok", questions, examEnabled: getExamStatus() }; } else if (action === "startSession") { const name = e.parameter.name || ""; if (!name) { result = { status:"error", message:"No name" }; } else if (!getExamStatus()) { result = { status:"disabled" }; } else { const sheet = getOrCreateSheet(ss, "Sessions", ["الاسم","وقت البداية (ms)","منتهي"]); const now = Date.now(); // Check if session exists and not expired const data = sheet.getDataRange().getValues(); let found = false; for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === name.trim()) { const startMs = Number(data[i][1]); const expired = data[i][2] === true || data[i][2] === "TRUE"; if (!expired && (now - startMs) < EXAM_DURATION_MS) { result = { status:"ok", startTime: startMs, serverNow: now, remaining: Math.max(0, EXAM_DURATION_MS - (now - startMs)) }; found = true; } else if (expired || (now - startMs) >= EXAM_DURATION_MS) { result = { status:"expired" }; found = true; } break; } } if (!found) { sheet.appendRow([name, now, false]); result = { status:"ok", startTime: now, serverNow: now, remaining: EXAM_DURATION_MS }; } } } else if (action === "checkTime") { const name = e.parameter.name || ""; const sheet = ss.getSheetByName("Sessions"); if (!sheet) { result = { status:"error" }; } else { const data = sheet.getDataRange().getValues(); const now = Date.now(); let found = false; for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === name.trim()) { const startMs = Number(data[i][1]); const expired = data[i][2] === true || data[i][2] === "TRUE"; const remaining = Math.max(0, EXAM_DURATION_MS - (now - startMs)); result = { status: (expired || remaining === 0) ? "expired" : "ok", remaining, serverNow: now }; found = true; break; } } if (!found) result = { status:"error", message:"Session not found" }; } } else if (action === "saveResult") { const name = e.parameter.name || ""; const total = e.parameter.total || 0; const correct = e.parameter.correct || 0; const score = e.parameter.score || "0%"; const startTime = e.parameter.startTime || ""; const endTime = e.parameter.endTime || ""; if (!name) { result = { status:"error", message:"No name" }; } else { const resSheet = getOrCreateSheet(ss, "Results", ["الاسم","إجمالي الأسئلة","الإجابات الصحيحة","النتيجة","وقت البدء","وقت الانتهاء"]); resSheet.appendRow([name, total, correct, score, startTime, endTime]); // Mark session as done const sessSheet = ss.getSheetByName("Sessions"); if (sessSheet) { const data = sessSheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === name.trim()) { sessSheet.getRange(i+1, 3).setValue(true); break; } } } result = { status:"ok" }; } } else if (action === "getResults") { const sheet = ss.getSheetByName("Results"); if (!sheet) { result = { status:"ok", results:[] }; } else { const data = sheet.getDataRange().getValues(); const results = []; for (let i = 1; i < data.length; i++) { if (!data[i][0]) continue; results.push({ name:data[i][0], total:data[i][1], correct:data[i][2], score:data[i][3], startTime:data[i][4], endTime:data[i][5] }); } result = { status:"ok", results }; } } else if (action === "getStatus") { result = { status:"ok", examEnabled: getExamStatus() }; } else if (action === "toggleExam") { const pass = e.parameter.pass || ""; if (pass !== PASS) { result = { status:"error", message:"Unauthorized" }; } else { const prop = PropertiesService.getScriptProperties(); const current = prop.getProperty("examEnabled") !== "false"; prop.setProperty("examEnabled", (!current).toString()); result = { status:"ok", examEnabled: !current }; } } else { result = { status:"error", message:"Unknown action" }; } } catch(err) { result = { status:"error", message: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function getExamStatus() { return PropertiesService.getScriptProperties().getProperty("examEnabled") !== "false"; } function getOrCreateSheet(ss, name, headers) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); } return sheet; }
